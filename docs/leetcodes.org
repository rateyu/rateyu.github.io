public int maxSubArray(int[] nums) {
        int max = nums[0];
        for(int i=0;i<nums.length;i++){
            int tmpMax = nums[i];
            if(tmpMax>max) {
                max = tmpMax;
            }
         for(int j=i+1;j<nums.length;j++) {
            tmpMax += nums[j];
            if(tmpMax>max) {
                max = tmpMax;
            }
         }
        }
        System.out.println("max--"+max);
        return max;
    }
